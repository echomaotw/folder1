def series_reshape(data, lag=2, n_pred=1, dropnan=True, var=['var1','var2']):
    df=pd.DataFrame(data)
    cols, names=list(), list()
    n_vars=df.shape[1]
    #x values in form of x-3,x-2,x-1
    for i in range(lag,0,-1):
        cols.append( df.shift(i) )
        names += [( var[j] + '(t-%d)' %(i) ) for j in range(n_vars)]
    #y values in form of y, y+1, y+2
    for i in range(0, n_pred):
        cols.append(df.shift(-i))
        if i== 0:
            
            names += [(var[j]+'(t)') for j in range(n_vars)]
        else:
            names += [(var[j]+'(t+%d)' %(i) ) for j in range(n_vars)]
    agg = pd.concat(cols, axis=1)
    agg.columns=names
    #drop na values
    if dropnan:
        agg.dropna(inplace=True)
    return agg
